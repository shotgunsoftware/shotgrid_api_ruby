def self.valid?(
  client_id: nil,
  client_secret: nil,
  username: nil,
  password: nil,
  session_token: nil,
  refresh_token: nil
)
  !!(
    (client_id && client_secret) || (password && username) ||
      session_token || refresh_token
  )
end
